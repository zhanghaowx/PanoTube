Panorama in Cube

Images are mapped to the surface of a cube.

                    
requirejs(['app'], function() {
    $(".panorama").panorama({
        type: "cube",
        tiles: HERE.Panorama.getImageUrls("1027084269", HERE.Panorama.RESOLUTION.HIGH)
    });
});
